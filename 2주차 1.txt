class Dog
{
private:
	int num;
public:
	Dog(int n) : num(n) {}
	friend std::ostream& operator<<(std::ostream& os, Dog n);
};

std::ostream& operator<<(std::ostream& os, Dog n)
{
	os << n.num;
	return os;
}

template <typename T>
void change(T& a, T& b)
{
	T temp{ a };
	a = b;
	b = temp;
}

//-------------
int main(void)
//-------------
{
	Dog a{ 1 }, b{ 2 };

	change(a, b);

	std::cout << a << ", " << b << '\n';	// 2, 1

	save("STL.cpp");
}
